#include <stdio.h>


int main()
{
    
int s=0;
    
int i=0;
    
for (i=1; i<1000; i=i + 4){
        
printf("число от 1 до 1000, увеличивающееся на 4:  %d\n",i);
        
if(i%7==0){
            
s=s+i;
        
}
    
}
    
printf("Суммируем все числа, которые делятся на 7 без остатка: %d\n",s);

    
return 0;

}
